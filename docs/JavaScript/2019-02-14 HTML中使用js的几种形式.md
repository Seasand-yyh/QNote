# HTML中使用js的几种形式

---



~~~html
<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<title>Document</title>
    <!-- 通过script标签引入外部js文件 -->
	<script type="text/javascript" src="main.js"></script>
    <!-- 在script标签内嵌入js代码 -->
    <script type="text/javascript">
		console.log('hello world')
	</script>
</head>
<body>
    <!-- 在a标签的href属性中执行js代码 -->
	<a href="javascript:func();">链接</a>
    <!-- 在form标签的action属性中执行js代码 -->
    <form action="javascript:func();">
        <input type="submit" value="提交"/>
    </form>
    <!-- 在事件中执行js代码 -->
    <div onclick="func();"></div>
</body>
</html>
~~~
