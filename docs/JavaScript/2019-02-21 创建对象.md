# 创建对象

---



~~~javascript
function createPerson(name, age){
    var person = new Object();
    person.name = name;
    person.age = age;
    person.say = function(){
        console.log('Hi, this is ' + this.name);
    }
    return person;
}
~~~



~~~javascript
function Person(name, age){
    this.name = name;
    this.age = age;
    this.say = function(){
        console.log('Hi, this is ' + this.name);
    }
}
~~~
