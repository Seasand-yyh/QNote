4、scrollLeft、scrollTop

页面滚动超出的距离。



5、

~~~javascript
//scroll 的兼容性处理
function scroll(){
    return {
        top: document.documentElement.scrollTop || document.body.scrollTop,
        left: document.documentElement.scrollLeft || document.body.scrollLeft
    };
}

//当页面出现滚动条时，page调整
function page(e){
    return {
        x: e.clientX + scroll().left,
        y: e.clientY + scroll().top
    };
}

~~~
